```typescript
import { Client } from '@modelcontextprotocol/sdk/client';

async function runTest() {
  console.log('Starting MCP client test...');
  
  // Create MCP client
  const client = new Client({
    serverUrl: 'http://localhost:3000/mcp'
  });
  
  try {
    // Initialize session
    console.log('Initializing session...');
    await client.initialize();
    console.log('Session initialized with ID:', client.sessionId);
    
    // List available tools
    console.log('\nListing available tools...');
    const tools = await client.listTools();
    console.log('Available tools:', tools);
    
    // Call imagine tool
    console.log('\nCalling imagine tool...');
    const imagineResult = await client.callTool('imagine', {
      type: 'string',
      length: 15
    });
    console.log('Imagine result:', imagineResult);
    
    // Call reveal tool
    console.log('\nCalling reveal tool...');
    const revealResult = await client.callTool('reveal', {
      format: 'json'
    });
    console.log('Reveal result:', revealResult);
    
    // Verify that reveal returned the same data
    console.log('\nVerifying results match:', imagineResult === revealResult ? 'SUCCESS' : 'FAILURE');
    
    // Clean up session
    console.log('\nCleaning up session...');
    await client.close();
    console.log('Session closed');
  } catch (error) {
    console.error('Error during test:', error);
  }
}

runTest();
```

# Test Strategy:
Run the client against the server and verify that it successfully completes the test flow. Check that the client correctly maintains the session ID between requests and that the reveal tool returns the same data that was generated by the imagine tool.
